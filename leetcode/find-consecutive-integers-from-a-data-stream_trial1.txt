class DataStream:

    def __init__(self, value: int, k: int):
        self.queue = deque()
        self.freq = Counter()
        self.value = value
        self.k = k

    def consec(self, num: int) -> bool:
        self.queue.append(num)
        self.freq[num] += 1
        while len(self.queue) > self.k:
            self.freq[self.queue[0]] -= 1
            self.queue.popleft()
        return self.freq[self.value] == self.k
