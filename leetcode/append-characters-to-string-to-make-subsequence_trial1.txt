class Solution:
    def appendCharacters(self, s: str, t: str) -> int:
        ans = 0
        i, j = 0, 0
        while i < len(s) and j < len(t):
            if s[i] == t[j]:
                i, j = i + 1, j + 1
                ans += 1
            else:
                i += 1
        return len(t) - ans
