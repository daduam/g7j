class Solution:
    def isPalindrome(self, s: str) -> bool:
        slow = ''.join([ch if ch.isdigit() or ch.isalpha() else '' for ch in s.lower()])
        i, j = 0, len(slow) - 1
        while i < j:
            if slow[i] != slow[j]:
                return False
            i, j = i + 1, j - 1
        return True
