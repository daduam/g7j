class Solution:
    def findMaxAverage(self, nums: List[int], k: int) -> float:
        psum = [0] * (len(nums)+1)
        for i in range(len(nums)):
            psum[i+1] = psum[i] + nums[i]
        i, ans = 0, -float('inf')
        while i+k-1 < len(nums):
            curavg = (psum[i+k] - psum[i]) / k
            ans = max(ans, curavg)
            i += 1
        return ans
