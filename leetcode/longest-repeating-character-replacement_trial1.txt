class Solution:
    def characterReplacement(self, s: str, k: int) -> int:
        ch2freq = defaultdict(int)
        left, ans = 0, 0
        for right in range(len(s)):
            ch2freq[s[right]] += 1
            while left <= right and right - left - max(ch2freq.values()) >= k:
                ch2freq[s[left]] -= 1
                left += 1
            ans = max(ans, right - left + 1)
        return ans
