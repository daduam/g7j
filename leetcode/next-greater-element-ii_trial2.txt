class Solution:
    def nextGreaterElements(self, nums: List[int]) -> List[int]:
        n = len(nums)
        ans = [-1] * (n*2)
        stack = []
        for i in range((len(nums) * 2)-1, -1, -1):
            curi = i % n
            while stack and stack[-1] <= nums[curi]:
                stack.pop()
            ans[i] = stack[-1] if stack else -1
            stack.append(nums[curi])
        return ans[:n]
