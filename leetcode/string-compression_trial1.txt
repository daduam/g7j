class Solution:
    def compress(self, chars: List[str]) -> int:
        holder = 0
        count = 1
        prev = chars[0]
        for i in range(1, len(chars)):
            if chars[i] == prev:
                count += 1
                continue
            chars[holder] = prev
            holder += 1
            prev = chars[i]
            if count > 1:
                for ch in str(count):
                    chars[holder] = ch
                    holder += 1
            count = 1
        if count > 0:
            chars[holder] = prev
            holder += 1
            if count > 1:
                for ch in str(count):
                    chars[holder] = ch
                    holder += 1
        return holder
